## Installation

```package-install
npx shadcn@latest add https://bracketcore.dev/r/match-card.json
```

## Prerequisites

Define your match data with teams, scores, and status.

```tsx
const match: Match = {
  id: "demo",
  round: 0,
  position: 0,
  bestOf: 3,
  status: "completed",
  teams: [
    {
      team: { id: "spirit", name: "Spirit" },
      score: 2,
      isWinner: true,
    },
    {
      team: { id: "falcons", name: "Falcons" },
      score: 0,
      isWinner: false,
    },
  ],
};
```

## Usage

```tsx
import { MatchCard, type Match } from "@bracketcore/registry";
```

```tsx
<MatchCard match={match} />
```

## Examples

### Live

A live match shows a destructive border, a pulsing live indicator in the header, and colored scores — green for the leading team, red for the trailing team.

<ComponentPreview code={`const live: Match = {
  id: "demo-live",
  round: 0,
  position: 0,
  bestOf: 5,
  status: "live",
  scheduledAt: todayAt(14, 0),
  teams: [
    {
      team: { id: "evil-geniuses", name: "Evil Geniuses" },
      score: 2,
    },
    {
      team: { id: "team-secret", name: "Team Secret" },
      score: 1,
    },
  ],
};

function todayAt(hours: number, minutes: number) {
  const d = new Date();
  d.setHours(hours, minutes, 0, 0);
  return d;
}

<MatchCard match={live} />
`}>
<MatchCardLive />
</ComponentPreview>

### Upcoming

An upcoming match displays the scheduled time and relative day in the header. The format adapts automatically: "Today", "Tomorrow", weekday name (within 7 days), or "Month Day".

<ComponentPreview code={`const upcoming: Match = {
  id: "demo-upcoming",
  round: 0,
  position: 0,
  bestOf: 3,
  status: "upcoming",
  scheduledAt: daysFromNow(1, 18, 0),
  teams: [
    {
      team: { id: "navi", name: "NaVi" },
      score: 0,
    },
    {
      team: { id: "virtus-pro", name: "Virtus.pro" },
      score: 0,
    },
  ],
};

function daysFromNow(days: number, hours: number, minutes: number) {
  const d = new Date();
  d.setDate(d.getDate() + days);
  d.setHours(hours, minutes, 0, 0);
  return d;
}

<MatchCard match={upcoming} />
`}>
<MatchCardUpcoming />
</ComponentPreview>

## Props

### `MatchCard`

| Prop           | Type                     | Description                                                          |
| -------------- | ------------------------ | -------------------------------------------------------------------- |
| `match`        | `Match`                  | The match data to display.                                           |
| `className`    | `string`                 | Additional CSS classes for the root element.                         |
| `onMatchClick` | `(match: Match) => void` | Callback fired when the card is clicked. Adds hover styles when set. |

### `Match`

```ts
interface Match {
  id: string;
  round: number;
  position: number;
  bestOf?: number;
  scheduledAt?: Date | string;
  status?: "upcoming" | "live" | "completed";
  teams: [MatchTeam, MatchTeam];
}
```

### `MatchTeam`

```ts
interface MatchTeam {
  team: Team | null;
  score: number;
  isWinner?: boolean;
}
```

### `Team`

```ts
interface Team {
  id: string;
  name: string;
  logo?: string;
  seed?: number;
}
```

## Theming

Colors use your shadcn theme variables directly (`--card`, `--border`, `--card-foreground`, `--muted-foreground`, `--primary`). No extra configuration needed.

The card width can be customized via a CSS custom property:

| Property                | Default | Description |
| ----------------------- | ------- | ----------- |
| `--bracket-match-width` | `13rem` | Card width. |

```tsx
<div style={{ "--bracket-match-width": "16rem" }}>
  <MatchCard match={match} />
</div>
```
