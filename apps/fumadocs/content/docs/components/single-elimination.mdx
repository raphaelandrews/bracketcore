<ComponentPreview code={`import { SingleElimination, type SingleEliminationBracket, type Match } from "@bracketcore/registry";


## Connector Styles

You can customize the connector style using the `connectorStyle` prop. Available options are `default` (standard bracket lines) and `simple` (squared lines).

## Usage

```tsx
import { SingleElimination, type SingleEliminationBracket } from "@bracketcore/registry";

const bracket: SingleEliminationBracket = {
  type: "single-elimination",
  rounds: [
    {
      name: "Semi-finals",
      matches: [
        {
          id: "sf-1",
          round: 0,
          position: 0,
          bestOf: 3,
          status: "completed",
          teams: [
            { team: { id: "a", name: "Team Alpha" }, score: 2, isWinner: true },
            { team: { id: "b", name: "Team Bravo" }, score: 1, isWinner: false },
          ],
        },
        {
          id: "sf-2",
          round: 0,
          position: 1,
          bestOf: 3,
          status: "completed",
          teams: [
            { team: { id: "c", name: "Team Charlie" }, score: 0, isWinner: false },
            { team: { id: "d", name: "Team Delta" }, score: 2, isWinner: true },
          ],
        },
      ],
    },
    {
      name: "Final",
      matches: [
        {
          id: "f-1",
          round: 1,
          position: 0,
          bestOf: 5,
          status: "completed",
          teams: [
            { team: { id: "a", name: "Team Alpha" }, score: 3, isWinner: true },
            { team: { id: "d", name: "Team Delta" }, score: 1, isWinner: false },
          ],
        },
      ],
    },
  ],
};

export default function Page() {
  return <SingleElimination bracket={bracket} />;
}
```

## Props

### `SingleElimination`

| Prop           | Type                       | Description                                     |
| -------------- | -------------------------- | ----------------------------------------------- |
| `bracket`        | `SingleEliminationBracket` | The bracket data containing rounds and matches.                                                |
| `className`      | `string`                   | Additional CSS classes for the root element.                                                   |
| `onMatchClick`   | `(match: Match) => void`   | Callback fired when a match card is clicked.                                                   |
| `connectorStyle` | `"default" \| "simple"`    | Connector line style. `"default"` uses curved/merged lines, `"simple"` uses squared-off lines. |

### `SingleEliminationBracket`

```ts
interface SingleEliminationBracket {
  type: "single-elimination";
  rounds: Round[];
}
```

### `Round`

```ts
interface Round {
  name: string;
  matches: Match[];
}
```

### `Match`

```ts
interface Match {
  id: string;
  round: number;
  position: number;
  bestOf?: number;
  scheduledAt?: string;
  status?: "upcoming" | "live" | "completed";
  teams: [MatchTeam, MatchTeam];
}
```

### `MatchTeam`

```ts
interface MatchTeam {
  team: Team | null;
  score: number;
  isWinner?: boolean;
}
```

### `Team`

```ts
interface Team {
  id: string;
  name: string;
  logo?: string;
  seed?: number;
}
```

## Sizing

Control dimensions via CSS custom properties on a parent element:

| Property                 | Default               | Description                                             |
| ------------------------ | --------------------- | ------------------------------------------------------- |
| `--bracket-match-width`  | `13rem`               | Width of each match card.                               |
| `--bracket-match-height` | `calc(3.25rem + 1px)` | Height of each match card (rows only, excludes header). |
| `--bracket-round-gap`    | `3rem`                | Horizontal gap between rounds.                          |
| `--bracket-match-gap`    | `1rem`                | Base vertical gap between matches in round 0.           |

```tsx
<div
  style={{
    "--bracket-match-width": "16rem",
    "--bracket-match-height": "5rem",
    "--bracket-round-gap": "4rem",
    "--bracket-match-gap": "1.5rem",
  }}
>
  <SingleElimination bracket={bracket} />
</div>
```
